SEGMENT data
value1: DB 5         ; байт
value2: DW 0x1234    ; слово
ENDS

SEGMENT code
ORG 100h

start:  MOV AL, value1     ; AL ← [value1]
        MOV BL, AL         ; BL ← AL
        OR  AL, BL         ; AL |= BL
        MUL AL             ; AX ← AL * AL
        JS  error          ; Прыжок по знаку, если результат отрицательный

        MOV AX, value2     ; AX ← [value2]
        MOV value2, AX     ; [value2] ← AX

        JP done            ; Прыжок при положительном флаге

error:  MOV AL, 0          ; AL ← 0

done:   MOV AH, 4Ch        ; DOS функция выхода
        INT 21h            ; Выход в DOS

ENDS
END
